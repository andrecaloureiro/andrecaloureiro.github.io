3. Relógio em Tempo Real

// Código do Relógio
function Relogio() {
  const [hora, setHora] = React.useState(new Date().toLocaleTimeString());

  React.useEffect(() => {
    const interval = setInterval(() => {
      setHora(new Date().toLocaleTimeString());
    }, 1000);
    return () => clearInterval(interval);
  }, []);

  return 
Hora Atual: {hora}
;
}
ReactDOM.createRoot(document.getElementById('relogio')).render();
